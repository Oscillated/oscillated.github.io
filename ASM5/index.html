白洪欢《汇编语言程序设计基础》笔记5

graphic mode

ptr

部分见4.

另外还有两种ptr用来修饰标号

near ptr 近，当前指令与目标地址在同一个段内

far ptr 远，当前指令与目标地址不在同一个段内

near ptr 和 far ptr的前面通常是jmp或call

  • jmp near ptr next
  • jmp far ptr away

标号 —— main: next: code段内

变量名 —— 无冒号 data段内

当目标操作数与源操作数的其中之一有明确的宽度时，另外一方不需要加类型修饰

当源操作数为常数，目标操作数为变量时，则该变量必须加上类型修饰

mov ds:[bx], 1; 错误
mov byte ptr ds:[bx], 1; 正确
mov word ptr ds:[bx], 1; 正确
mov dword ptr ds:[bx], 1; 正确

部分见4.

图形模式

通过写入B800段内的显卡内存来控制屏幕字符输出是文本模式编程

映射机制mapping：把显卡内存和RAM中的B800段建立对应关系。

A000段开始到F000段均不是用户的内存空间，部分内存映射到显卡内存中，部分内存映射到ROM中。

POST(Power on self test)位于ROM中，映射到F000段中，只读、不能更改

DOS中留给用户的内存空间只有0000:0000-9000:FFFF 640k内存

  • 显卡有两种模式 图像模式G（13h） 文本模式T(14h)

  • 把显卡切换到图形模式

    • int 10h中断

    • ;切换到分辨率320*200，颜色为256色的图形模式
mov ah, 0      ; 其中AH=0表示int 10h的子功能号为0
mov al, 13h; 其中AL=13h代表图形模式编号 AL=03h为80*25文本模式
int 10h     ; int 10h是与显示相关的bios中断

; 输出红色实心方块
code segment
assume cs:code; cs不需要赋值会自动等于code
main:
jmp begin
i  dw 0 ;i定义在code segment中，这样操作，是避免让变量的值被当作指令执行，经编译时dw是对编译器的命令，是伪指令，经编译后只有dw后面的0才会变成机器码
;编译时处理对cpu的命令和对编译器的命令是顺序执行的，遇到cpu指令时编译成机器码，遇到dw伪指令时只把变量值生成机器码。编译并不是执行，所以语句都要编译
begin:
mov ax, 0013h ;转到图形模式
int 10h
mov ax, 0A000h
mov es, ax
;(320/2, 200/2)
mov di, (100-20)*320+(160-20); (160-20,100-20)在屏幕中间位置画，为此要先得到左上角第一个点所对应的偏移地址
           ;mov cx, 41; rows=41
mov i, 41
next_row:
           ;push cx
push di
mov al, 4; color=red
mov cx, 41; dots=41 41是列数，每行画41个点
next_dot:
mov es:[di], al ; next_dot的循环画一整行的点
add di, 1
sub cx, 1
jnz next_dot
pop di; 取出保存的左上角(x,y)对应的地址
            ;pop cx 取出保存的行数继续用 cx=41
add di, 320; 下一行的起点的地址，因为上下差320个字节
            ;sub cx, 1; 行数-1
sub i, 1
jnz next_row
mov ah,0
int 16h;bios键盘输入,类似int 21h的01h功能
mov ax, 0003h
int 10h; int 10h 3号功能，切换到80*25文本模式
; int 10h 0号功能为视频显示模式
; 未切换的话，字体会变大，每行元素个数减少不足80个
mov ah, 4Ch
int 21h
code ends
end main

; 显示汉字“我”点阵
data segment
hz db 04h,80h,0Eh,0A0h,78h,90h,08h,90h ;每两个16进制数据代表一行，一行16位分成8+8由两个数据分别代表写0和1的情况
db 08h,84h,0FFh,0FEh,08h,80h,08h,90h
db 0Ah,90h,0Ch,60h,18h,40h,68h,0A0h
db 09h,20h,0Ah,14h,28h,14h,10h,0Ch
data ends
code segment
assume cs:code, ds:data
main:
mov ax, data
mov ds, ax ; ds代码段
mov ax, 0A000h
mov es, ax ; es显卡内存空间
mov di, 0
mov ax, 0013h ; 开图形模式
int 10h
mov dx, 16
mov si, 0
next_row:
mov ah, hz[si] ; 每行的操作就是两个数据写入
mov al, hz[si+1] 
add si, 2
mov cx, 16
check_next_dot:
shl ax, 1; 左移移出的位会自动进入CF(进位标志)
jnc no_dot; 若没有进位即CF=0则跳到no_dot
is_dot:
mov byte ptr es:[di], 0Ch ;这一位有像素点，则用0C（高亮度红色）上色
no_dot:
add di, 1
sub cx, 1
jnz check_next_dot
sub di, 16
add di, 320
sub dx, 1
jnz next_row
mov ah, 1
int 21h
mov ax, 0003h
int 10h
mov ah, 4Ch
int 21h
code ends
end main

文本形式的“我”（用*构成点阵）
data segment
hz db 04h,80h,0Eh,0A0h,78h,90h,08h,90h
db 08h,84h,0FFh,0FEh,08h,80h,08h,90h
db 0Ah,90h,0Ch,60h,18h,40h,68h,0A0h
db 09h,20h,0Ah,14h,28h,14h,10h,0Ch
data ends
code segment
assume cs:code, ds:data
main:
mov ax, data
mov ds, ax
mov ax, 0B800h
mov es, ax
mov ax, 0003h
int 10h
mov di, 0
mov dx, 16
mov si, 0
next_row:
mov ah, hz[si]
mov al, hz[si+1]
add si, 2
mov cx, 16
check_next_dot: ;每次取两个字节判断16位中是否有1，有1则用*画点
shl ax, 1 ;某一位有1时（即左移移出的位是1，这一位会进入CF）
jnc no_dot ;移出的位是1，说明有进位，用*画点
is_dot:
mov byte ptr es:[di], '*'
mov byte ptr es:[di+1], 0Ch
no_dot:
add di, 2
sub cx, 1
jnz check_next_dot
sub di, 32
add di, 160
sub dx, 1
jnz next_row
mov ah, 1
int 21h
mov ah, 4Ch
int 21h
code ends
end main
